// Creating tables
Table users as U {
  id int [pk, increment] // auto-increment
  full_name varchar
  user_name varchar
  pasword varchar
  country_code int
  birthaday datetime
  img_path path
}



Table review {
  user_id int [pk]
  film_id int [pk]
  stars double
}

Table countries {
  code int [pk]
  name varchar
 }

// Creating references
// You can also define relaionship separately
// > many-to-one; < one-to-many; - one-to-one
Ref: U.id > review.user_id
Ref: film.id > review.film_id
Ref: U.country_code < countries.code

Ref: actor.id > role.actor_id
